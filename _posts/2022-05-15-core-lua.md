# ***vorp core is the master tool for all your projects***

# ***USAGE***

* Calling the API for Notifications

```lua
-- at the top of your server side files
local VORPcore = {}

TriggerEvent("getCore", function(core)
    VORPcore = core
end)

```

* The same Event can be called from client side to use `different functions`
  * for Notifications
  * for CallBacks


# ***Client side***

* Notify

* hide or show CORE UI

```lua
    -------------------------------------------------
    -- from server side
    local _source = source

    TriggerClientEvent("vorp:showUi", _source, false) 

    -------------------------------------------------
    -- from client side
    TriggerEvent("vorp:showUi", true) 

```

# Server API 

* Get max characters 

```lua
-- at the top of the server file
local VORPcore = {}

TriggerEvent("getCore", function(core)
    VORPcore = core
end)

-- this returns a number
local maxChars = VORPcore.maxCharacters 

```

* Get Users data

```lua
-- at the top of the server file
local VORPcore = {}

TriggerEvent("getCore", function(core)
    VORPcore = core
end)

local _source = source
-- contains functions and information from all characters
local User = VORPcore.getUser(_source)

-- Return user group (not character group)
local UserGroup = User.getGroup 

-- Return character selected by user
local Character = User.getUsedCharacter 
-- or
local Character = VORPcore.getUser(_source).getUsedCharacter

--Data you can get
Character.identifier
Character.charIdentifier
Character.group
Character.job
Character.jobGrade
Character.money
Character.gold
Character.rol
Character.xp
Character.firstname
Character.lastname
Character.inventory
Character.status
Character.coords
Character.isdead
Character.skin
Character.comps

```

* Set functions 

```lua
-- at the top of the server file
local VORPcore = {}

TriggerEvent("getCore", function(core)
    VORPcore = core
end)

local _source = source

local Character = VORPcore.getUser(_source).getUsedCharacter

--Functions you can set using the API

Character.setJob("miner")
Character.setJobGrade(1)
Character.setGroup("admin")
Character.setMoney(1000.50)
Character.setGold(1000.20)
Character.setRol(1000)
Character.setXp(5000)
Character.setFirstname("Sadie")
Character.setLastname("Adler")
Character.updateSkin("need comps in json")
Character.updateComps("need comps in json")
Character.addCurrency(0, 1000) -- Add money 1000 | 0 = money, 1 = gold, 2 = rol
Character.removeCurrency(0, 1000) -- Remove money 1000 | 0 = money, 1 = gold, 2 = rol
Character.addXp(100)
Character.removeXp(100)

```

* You can also use Trigger Events from server side only


```lua
-- server side
local _source = source

TriggerEvent("vorp:getCharacter",_source,function(user)
 
    print(user.charIdentifier)
    print(user.group)
    print(user.job)
    print(user.jobGrade)
    print(user.money)
    print(user.gold)
    print(user.rol)
    print(user.xp)
    print(user.firstname)
    print(user.lastname)
    print(user.inventory)
    print(user.status)
    print(user.coords)
    print(user.isdead)
    print(user.skin)
    print(user.comps) 

end)
```

* Trigger Events from server side only

```lua
-- server side
local _source = source
--set job
TriggerEvent("vorp:setJob", _source, "anything")

-- currency
TriggerEvent("vorp:addMoney", _source, 0, 1000.50) -- add 1000.50 dollars to source
TriggerEvent("vorp:addMoney", _source, 1, 1000.50) -- add 1000.50 gold to source
TriggerEvent("vorp:addMoney", _source, 2, 1000) -- add 1000 rol to source

TriggerEvent("vorp:removeMoney", _source, 0, 1000.50) -- remove 1000.50 dollars from source
TriggerEvent("vorp:removeMoney", _source, 1, 1000.50) -- remove 1000.50 gold from source
TriggerEvent("vorp:removeMoney", _source, 2, 1000) -- remove 1000 role from source

TriggerEvent("vorp:addXp", _source, 100) -- add 100 xp to source
TriggerEvent("vorp:removeXp", _source, 100) -- remove 100 xp from source

```
